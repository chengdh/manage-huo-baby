        def initialize
                fn = File.join(File.dirname(File.expand_path(__FILE__)),'dict/Mandarin.dat')
                @codes = {}
                File.readlines(fn).each do |line|
                        nv = line.split(/\s/)
                        @codes[nv[0]] = nv[1]
                end
        end
